    6. Create `Tools/SkillGuideTools.cs` -- MCP tool class:
       ```csharp
       [McpServerToolType]
       public class SkillGuideTools
       {
           [McpServerTool(Name = "get-skill-guide", Description = "Returns the QdrantSkillsMCP usage guide that teaches agents how to effectively use this server")]
           public string GetSkillGuide()
           {
               var assembly = typeof(SkillGuideTools).Assembly;
               using var stream = assembly.GetManifestResourceStream(
                   "QdrantSkillsMCP.Infrastructure.Skill.SKILL.md");
               if (stream is null)
                   throw new InvalidOperationException("SKILL.md embedded resource not found");
               using var reader = new StreamReader(stream);
               return reader.ReadToEnd();
           }
       }
       ```
